def pdf(x, mean, cov,k):
    
    x = x.reshape(k,1)
    mean = mean.reshape(k,1)
    first_term = 1/(np.sqrt(2*np.pi**k)* (np.linalg.det(cov)))
    #print((np.linalg.det(cov).shape))
    
    second_term = np.exp(-0.5* (((x-mean).T).dot(np.linalg.inv(cov))).dot(x-mean))
    
    return(first_term *second_term)

#Best Path Matrix

for i in range(1, P_dashv.shape[1]):
    
    for k in range(0,2):
        B[k,i] = np.argmax(T[:,k] * P_dashv[:, i-1])
    for k in range(0,2):
        P_dashv[k,i] = T[B[k,i], k] * P_dashv[B[k,i], i-1] * pdf[k, i]
    
    #Normalizing probabilities
    sum = P_dashv[0,i] + P_dashv[1,i]

num_iteration1 = 20000
error_sum = 0
error_fin = np.zeros((1,num_iteration1))

for i in range(num_iteration1):
    for j in range(152):
        #the forward propogation part
        t_inner = A.dot(np.concatenate((X[:,j], [1]), axis=0))
        logis_t = sigmoid(t_inner)
        fin_forward = A_out.dot(np.concatenate((logis_t, [1]), axis=0))
        y_hat = sigmoid(fin_forward)
        #print (y_hat.shape)
        #print (y[:,j].shape)
        error = (y[:,j] - y_hat)**2
        error_sum = error_sum + error
        
        #backward propogation
        dfrn_y_hat = np.multiply(y_hat,(1-y_hat))
        
        delt_2 = np.multiply((y_hat - y[:,j]),dfrn_y_hat)
        delta_A_out = delt_2.dot((np.concatenate((logis_t, [1]), axis=0)).reshape((4,1)).T)
        A_out = A_out - ro * delta_A_out
        
        #backpropogate to the inner layer, i.e we have to update A
        dfrn_logis_t = logis_t * (1 - logis_t)
